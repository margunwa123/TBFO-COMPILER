S -> DEF_COND S | WITH_COND S | WHILE_COND S | WITH_COND S | IF_COND S | IMPORT_COND S | FOR_COND S | RAISE_COND S | ALGORITHM S | EPSILON
DEF_COND -> DEF VARIABLE COLON STATEMENT RETURN_STATEMENT
WHILE_COND -> while BRACKET_STATEMENT COLON LOOP_STATEMENT
IF_COND -> IF BRACKET_STATEMENT COLON STATEMENT | IF BRACKET_STATEMENT COLON STATEMENT ELSE_STATEMENT
IMPORT_COND -> IMPORT VARIABLE | IMPORT VARIABLE AS VARIABLE
FOR_COND -> FOR VARIABLE IN VARIABLE COLON LOOP_STATEMENT | FOR VARIABLE IN RANGE INT COLON
WITH_COND -> with open BRACKET_STATEMENT as VARIABLE COLON
COLON -> :
DEF -> def
LOGICAL -> + | - | * | / | % 
RELATIONAL -> == | <= | >= | < | >
{ ini buat else statement yang dipake diatas }
ELSE_STATEMENT -> ELSE COLON STATEMENT | ELSE COLON IF_COND | ELIF_STATEMENT 
ELIF_STATEMENT -> ELIF BRACKET_STATEMENT COLON STATEMENT | ELIF_STATEMENT STATEMENT COLON IF_COND
BRACKET_STATEMENT -> OPEN_BRACKET STATEMENT CLOSE_BRACKET | STATEMENT
LOOP_STATEMENT -> LOOP_STATEMENT STATEMENT | 
RETURN_STATEMENT -> STATEMENT | STATEMENT RETURN VARIABLE
STATEMENT -> True | False | pass | VARIABLE RELATIONAL VARIABLE | NOT VARIABLE | RAISE_COND | continue |
